    34→  const {

    38→    notificationChannels, updateNotificationChannels,

    42→  
    43→  const { notificationsEnabled, showNotification } = useNotifications()
    44→  const [scanning, setScanning] = useState(false)
    45→  const [progress, setProgress] = useState({ scanned: 0, total: 0, found: 0 })
    46→  // Read from store (persisted) instead of separate localStorage
    47→  const [telegramBotToken, setTelegramBotToken] = useState(settings.telegramBotToken || '')
    48→  const [telegramChatId, setTelegramChatId] = useState(settings.telegramChatId || '')
    49→  const [whatsappPhone, setWhatsappPhone] = useState('')
    50→  const [emailAddress, setEmailAddress] = useState('')
    51→  const [showSetup, setShowSetup] = useState(false)
    52→  const scheduleRef = useRef<ReturnType<typeof setInterval> | null>(null)
    53→
    54→  // Sync local state if settings change (e.g., user updates in Telegram tab)
    55→  useEffect(() => {
    56→    setTelegramBotToken(settings.telegramBotToken || '')
    57→    setTelegramChatId(settings.telegramChatId || '')
    58→  }, [settings.telegramBotToken, settings.telegramChatId])
    59→
    60→  // Save telegram token to store when user types in Scanner
    61→  function saveTelegramToken() {
    62→    updateSettings({ 
    63→      telegramBotToken: telegramBotToken || null,
    64→      telegramChatId: telegramChatId || null,
    65→    })
    66→  }
    67→
    68→  // ---- Run scan (ENHANCED Phase 6) ----
    69→  const runScan = useCallback(async () => {
    70→    setScanning(true)
    71→    setProgress({ scanned: 0, total: 0, found: 0 })
    72→    
    73→    try {
    74→      // MTF config (all 7 timeframes)
    75→      const mtfConfig = {
    76→        timeframes: ['5m', '15m', '30m', '1H', '4H', '12H', '1D'] as ScanTimeframe[],
    77→        htfTimeframe: '1D' as ScanTimeframe,
    78→        ltfTimeframe: '1H' as ScanTimeframe,
    79→        requireHTFAlignment: false,
    80→        requireMTFAgreement: 4,
    81→      }
    82→      
    83→      const summary = await scanMultipleCoinsEnhanced(
    84→        scannerConfig,
    85→        mtfConfig,
    86→        settings.capital,
    87→        settings.riskPerTrade,
    88→        (scanned, total, found) => {
    89→          setProgress({ scanned, total, found })
    90→        }
    91→      )
    92→      
    93→      setLastScanSummary(summary)
    94→      updateScheduledScan({
    95→        lastScanAt: summary.scannedAt,
    96→        totalScansRun: scheduledScan.totalScansRun + 1,
    97→      })
    98→      
    99→      pushAlert({
   100→        type: 'INFO',
   101→        title: '🔍 Enhanced Scan Complete',
   102→        message: `${summary.totalScanned} coins scanned, ${summary.totalWithSignals} setups found (${summary.longSignals}L / ${summary.shortSignals}S) | Market: ${summary.marketRiskLevel} | Mode: ${summary.tradingMode}`,
   103→      })
   104→      
   105→      // Send notifications for top picks
   106→      if (summary.topPicks.length > 0) {
   107→        await sendNotifications(summary.topPicks.slice(0, 3))
   108→      }
   109→    } catch (err) {
   110→      console.error('Scan failed:', err)
   111→      pushAlert({
   112→        type: 'CRITICAL',
   113→        title: 'Scan Failed',
   114→        message: err instanceof Error ? err.message : 'Unknown error',
   115→      })
   116→    } finally {
   117→      setScanning(false)
   118→    }
   119→  }, [scannerConfig, scheduledScan, pushAlert, setLastScanSummary, updateScheduledScan, settings])
   120→
   121→  // ---- Send notifications ----
   122→  // Logic:
   123→  // - Telegram: ONLY for setups that passed deep analysis (6-layer validated) + smart recommendation BUY/STRONG_BUY
   124→  // - Browser Push: for all setups with confidence > threshold (broader)
   125→  // - WhatsApp/Email: for deep-passed only (semi-manual)
   126→  async function sendNotifications(picks: EnhancedScannerResult[]) {
   127→    let alertsSent = 0
   128→    let telegramSent = 0
   129→    let browserPushSent = 0
   130→    
   131→    for (const pick of picks) {
   132→      const deep = pick.deepAnalysis
   133→      const rec = pick.smartRecommendation
   134→      const isDeepPassed = deep?.passedDeepAnalysis === true
   135→      const isStrongConfluence = deep?.confluence === 'STRONG' || deep?.confluence === 'MEGA'
   136→      const isRecommended = rec?.finalRecommendation === 'STRONG_BUY' || rec?.finalRecommendation === 'BUY'
   137→      
   138→      // Telegram: ONLY for deep-passed + recommended (Phase 6: also check smart recommendation)
   139→      if (scannerConfig.notifyTelegram && telegramBotToken && telegramChatId && isDeepPassed && isRecommended) {
   140→        try {
   141→          // Use enhanced alert format (ultra detailed) if available
   142→          const message = rec && pick.mtfAnalysis && pick.eventIntelligence
   143→            ? formatEnhancedAlertMessage(pick)
   144→            : deep 
   145→            ? formatDeepAlertMessage(pick, deep, await getScanMacroData())
   146→            : formatAlertMessage(pick)
   147→          
   148→          const res = await fetch('/api/telegram/send', {
   149→            method: 'POST',
   150→            headers: { 'Content-Type': 'application/json' },
   151→            body: JSON.stringify({
   152→              botToken: telegramBotToken,
   153→              chatId: telegramChatId,
   154→              message,
   155→            }),
   156→          })
   157→          if (res.ok) {
   158→            alertsSent++
   159→            telegramSent++
   160→            console.log(`Telegram alert sent for ${pick.symbol} (score ${rec?.finalScore.toFixed(0)}, rec ${rec?.finalRecommendation})`)
   161→          }
   162→        } catch (err) {
   163→          console.error('Telegram send failed:', err)
   164→        }
   165→      }
   166→      
   167→      // Browser Push: for ALL setups with confidence > threshold (broader)
   168→      if (scannerConfig.notifyBrowserPush && notificationsEnabled) {
   169→        let title = `🔔 ${pick.name} ${pick.bestSignal!.bias} Signal`
   170→        let body = `${pick.bestSignal!.styleName} • Confidence ${pick.bestSignal!.confidence.toFixed(0)}%`
   171→        
   172→        if (rec?.finalRecommendation === 'STRONG_BUY') {
   173→          title = `⭐ ${pick.name} ${pick.bestSignal!.bias} — STRONG BUY!`
   174→          body = `${rec.tradeTypeLabel} • Score ${rec.finalScore.toFixed(0)}/100 • Hold: ${rec.estimatedHoldTime}`
   175→        } else if (rec?.finalRecommendation === 'AVOID') {
   176→          title = `⚠️ ${pick.name} — AVOID (event/conflict)`
   177→          body = `${rec.recommendationReason}`
   178→        } else if (isDeepPassed) {
   179→          title = `✅ ${pick.name} ${pick.bestSignal!.bias} — 6-Layer Validated`
   180→          body = `${pick.bestSignal!.styleName} • Score ${deep?.totalLayerScore.toFixed(0)}/100`
   181→        } else if (isStrongConfluence) {
   182→          title = `🔥 ${pick.name} ${pick.bestSignal!.bias} — Strong Confluence`
   183→          body = `${pick.bestSignal!.styleName} • Confluence: ${deep?.confluence}`
   184→        }
   185→        
   186→        showNotification(title, body, pick.symbol)
   187→        alertsSent++
   188→        browserPushSent++
   189→      }
   190→      
   191→      // WhatsApp (semi-auto): for deep-passed + recommended only
   192→      if (scannerConfig.notifyWhatsApp && whatsappPhone && isDeepPassed && isRecommended) {
   193→        const waMessage = formatWhatsAppMessage(pick)
   194→        const waUrl = `https://wa.me/${whatsappPhone}?text=${encodeURIComponent(waMessage)}`
   195→        window.open(waUrl, '_blank')
   196→        alertsSent++
   197→      }
   198→      
   199→      // Email (semi-auto): for deep-passed + recommended only
   200→      if (scannerConfig.notifyEmail && emailAddress && isDeepPassed && isRecommended) {
   201→        const subject = `🚨 ${pick.name} ${pick.bestSignal!.bias} — ${rec?.finalRecommendation || 'Signal'}`
   202→        const body = formatWhatsAppMessage(pick)
   203→        const mailtoUrl = `mailto:${emailAddress}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`
   204→        window.open(mailtoUrl, '_blank')
   205→        alertsSent++
   206→      }
   207→    }
   208→    
   209→    if (alertsSent > 0) {
   210→      updateScheduledScan({
   211→        totalAlertsSent: scheduledScan.totalAlertsSent + alertsSent,
   212→      })
   213→      pushAlert({
   214→        type: 'INFO',
   215→        title: '📊 Notification Summary',
   216→        message: `${telegramSent} Telegram (deep+recommended) + ${browserPushSent} Browser Push (all candidates)`,
   217→      })
   218→    }
   219→  }

   221→  // Helper to get macro data (cached in component)
   222→  const [cachedMacro, setCachedMacro] = useState<{ fearGreed: number; fearGreedLabel: string } | null>(null)

   231→    } catch {
   232→      return { fearGreed: 50, fearGreedLabel: 'Neutral' }
   233→    }
   234→  }

   244→      const intervalMs = scannerConfig.autoScheduleHours * 3600 * 1000

   247→        intervalHours: scannerConfig.autoScheduleHours,

   259→        nextScanAt: null,
   260→      })

   262→    
   263→    return () => {
   264→      if (scheduleRef.current) clearInterval(scheduleRef.current)
   265→    }

   289→          <div><strong>Telegram alerts:</strong> ONLY for 6-layer validated setups (high quality, low spam)</div>
   290→          <div><strong>Browser push:</strong> For all candidates (broader, you filter manually)</div>
   291→          <div><strong>Confluence:</strong> 2+ styles agree = STRONG, 3+ = MEGA (highest priority)</div>
   292→        </AlertDescription>
   293→      </Alert>
   294→
   295→      {/* Notification Channels Setup */}

   301→              Notification Channels

   307→          </div>
   308→        </CardHeader>
   309→        <CardContent className="space-y-2">
   310→          <div className="grid grid-cols-2 gap-2 text-xs">
   311→            <label className="flex items-center gap-2 p-2 rounded border border-zinc-800">
   312→              <Checkbox
   313→                checked={scannerConfig.notifyTelegram}
   314→                onCheckedChange={(v) => updateScannerConfig({ notifyTelegram: v === true })}
   315→              />
   316→              <Send className="h-3 w-3 text-blue-400" />
   317→              <span>Telegram</span>
   318→            </label>
   319→            <label className="flex items-center gap-2 p-2 rounded border border-zinc-800">
   320→              <Checkbox
   321→                checked={scannerConfig.notifyBrowserPush}
   322→                onCheckedChange={(v) => updateScannerConfig({ notifyBrowserPush: v === true })}
   323→              />
   324→              <Bell className="h-3 w-3 text-emerald-400" />
   325→              <span>Browser Push</span>
   326→            </label>
   327→            <label className="flex items-center gap-2 p-2 rounded border border-zinc-800">

   333→              <span>WhatsApp (link)</span>
   334→            </label>
   335→            <label className="flex items-center gap-2 p-2 rounded border border-zinc-800">
