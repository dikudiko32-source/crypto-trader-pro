    43→export function SetupFinder() {
    44→  const [symbol, setSymbol] = useState('BTCUSDT')
    45→  const [bias, setBias] = useState<Bias>('LONG')
    46→  const [loading, setLoading] = useState(false)
    47→  const [result, setResult] = useState<SetupResult | null>(null)
    48→  const [error, setError] = useState<string | null>(null)
    49→  const [allCoins, setAllCoins] = useState<CoinMarketData[]>([])
    50→  const [checklist, setChecklist] = useState(defaultChecklist())
    51→  const [openPositions, setOpenPositions] = useState(0)
    52→  const [dailyDD, setDailyDD] = useState(0)
    53→
    54→  const { settings, addSetup, pushAlert, setups } = useAppStore()
    55→
    56→  // Load top coins for autocomplete
    57→  useEffect(() => {
    58→    async function loadCoins() {
    59→      const coins = await getTopCoins(100)
    60→      setAllCoins(coins)
    61→    }
    62→    loadCoins()
    63→  }, [])
    64→
    65→  async function analyze() {
    66→    setLoading(true)
    67→    setError(null)
    68→    setResult(null)
    69→    try {
    70→      // Fetch data for all timeframes
    71→      const timeframeData: Record<Timeframe, TechnicalAnalysis> = {} as Record<Timeframe, TechnicalAnalysis>
    72→      const candlesByTf: Record<Timeframe, Candle[]> = {} as Record<Timeframe, Candle[]>
    73→      
    74→      await Promise.all(
    75→        TIMEFRAMES.map(async (tf) => {
    76→          const candles = await getKlines(symbol, tf, 300)
    77→          candlesByTf[tf] = candles
    78→          // Always analyze, even with few candles (analyzeTimeframe handles edge cases)
    79→          timeframeData[tf] = analyzeTimeframe(candles, tf)
    80→        })
    81→      )
    82→
    83→      // Check we have at least 1H and 4H data (the entry timeframes)
    84→      if (!timeframeData['1H'] || !timeframeData['4H'] || candlesByTf['4H'].length < 20 || candlesByTf['1H'].length < 20) {
    85→        throw new Error('Insufficient candle data for 1H/4H. Check symbol.')
    86→      }
    87→
    88→      // Fetch additional data
    89→      const [ticker, funding, oi, lsr] = await Promise.all([
    90→        getTicker24h(symbol),
    91→        getFundingRate(symbol),
    92→        getOpenInterest(symbol.replace('USDT', '') + 'USDT'),
    93→        getLongShortRatio(symbol.replace('USDT', '') + 'USDT'),
    94→      ])
    95→
    96→      // Find coin fundamental (from CoinGecko)
    97→      const coinData = allCoins.find(c => 
    98→        c.symbol + 'USDT' === symbol || c.symbol === symbol.replace('USDT', '')
    99→      )
   100→      
   101→      const fundamental: CoinFundamental = {
   102→        symbol: symbol.replace('USDT', ''),
   103→        name: coinData?.name || symbol,
   104→        marketCap: coinData?.marketCap || 0,
   105→        marketCapRank: coinData?.marketCapRank || 999,
   106→        volume24h: ticker?.quoteVolume || coinData?.totalVolume || 0,
   107→        volToMcap: coinData ? (ticker?.quoteVolume || coinData.totalVolume) / coinData.marketCap : 0,
   108→        exchangeListings: 8,
   109→        tokenomics: {
   110→          circulating: coinData?.circulatingSupply || 0,
   111→          total: coinData?.totalSupply || 0,
   112→          fdv: coinData && coinData.totalSupply ? coinData.currentPrice * coinData.totalSupply : 0,
   113→          fdvToMcap: coinData && coinData.totalSupply && coinData.marketCap > 0 
   114→            ? (coinData.currentPrice * coinData.totalSupply) / coinData.marketCap 
   115→            : 1,
   116→          nextUnlock: null,
   117→        },
   118→        activeAddresses30d: 0,
   119→        activeAddressesTrend: 'FLAT',
   120→        devActivity: 50,
   121→        newsSentiment: 50,
   122→        verdict: 'PASS',
   123→      }
   124→
   125→      // Determine setup type based on technical analysis
   126→      const daily = timeframeData['1D']
   127→      const h4 = timeframeData['4H']
   128→      
   129→      let setupType: SetupType = 'TREND_PULLBACK'
   130→      if (daily.adx < 20) setupType = 'MEAN_REVERSION'
   131→      else if (h4.bollinger.squeeze && daily.bollinger.squeeze) setupType = 'BREAKOUT'
   132→
   133→      // Calculate Fibonacci levels
   134→      const recentCandles = candlesByTf['4H'].slice(-50)
   135→      const swingLow = Math.min(...recentCandles.map(c => c.low))
   136→      const swingHigh = Math.max(...recentCandles.map(c => c.high))
   137→      const fib = fibonacci(swingLow, swingHigh)
   138→
   139→      // Calculate entry zone, SL, TP based on bias
   140→      const currentPrice = timeframeData['1H'].price
   141→      let entryZone: { lower: number; upper: number }
   142→      let stopLoss: number
   143→      let tps: { tp1: number; tp2: number; tp3: number }
   144→      let entryTrigger: string
   145→
   146→      if (bias === 'NEUTRAL') {
   147→        setError('Pilih LONG atau SHORT — NEUTRAL tidak bisa dianalisis')
   148→        return
   149→      }
   150→      if (bias === 'LONG') {
   151→        // Long setup: entry at pullback to fib 0.5-0.618 + EMA50 confluence
   152→        const fib618 = fib.levels['0.618']
   153→        const fib500 = fib.levels['0.500']
   154→        const ema50_4h = h4.ema50
   155→        entryZone = {
   156→          lower: Math.min(fib618, ema50_4h * 0.99),
   157→          upper: Math.max(fib500, ema50_4h * 1.01),
   158→        }
   159→        // SL below fib 0.786 or recent swing low
   160→        stopLoss = Math.min(fib.levels['0.786'], swingLow * 0.98)
   161→        // TPs at previous high + fib extensions
   162→        tps = {
   163→          tp1: swingHigh,
   164→          tp2: fib.extensions['1.272'],
   165→          tp3: fib.extensions['1.618'],
   166→        }
   167→        entryTrigger = `1H close above ${((entryZone.lower + entryZone.upper) / 2 * 1.005).toFixed(currentPrice < 1 ? 5 : 2)} with volume 1.5x avg`
   168→      } else if (bias === 'SHORT') {
   169→        // Short setup: entry at pullback to fib 0.5-0.618 (from highs)
   170→        const fib618 = fib.levels['0.618']
   171→        const fib500 = fib.levels['0.500']
   172→        const ema50_4h = h4.ema50
   173→        entryZone = {
   174→          lower: Math.min(fib500, ema50_4h * 0.99),
   175→          upper: Math.max(fib618, ema50_4h * 1.01),
   176→        }
   177→        stopLoss = Math.max(fib.levels['0.786'], swingHigh * 1.02)
   178→        tps = {
   179→          tp1: swingLow,
   180→          tp2: swingLow - (swingHigh - swingLow) * 0.272,
   181→          tp3: swingLow - (swingHigh - swingLow) * 0.618,
   182→        }
   183→        entryTrigger = `1H close below ${((entryZone.lower + entryZone.upper) / 2 * 0.995).toFixed(currentPrice < 1 ? 5 : 2)} with volume 1.5x avg`
   184→      }
   185→
   186→      const entryMid = (entryZone.lower + entryZone.upper) / 2
   187→      const risk = Math.abs(entryMid - stopLoss)
   188→      const tp1Rr = Math.abs(tps.tp1 - entryMid) / risk
   189→      const tp2Rr = Math.abs(tps.tp2 - entryMid) / risk
   190→      const tp3Rr = Math.abs(tps.tp3 - entryMid) / risk
   191→      const avgRr = (tp1Rr * 0.5 + tp2Rr * 0.3 + tp3Rr * 0.2)
   192→
   193→      // Position size
   194→      const riskAmount = settings.capital * (settings.riskPerTrade / 100)
   195→      const usdToIdr = 15800 // approximate
   196→      const usdRisk = riskAmount / usdToIdr
   197→      const quantity = usdRisk / risk
   198→      const nominalValue = quantity * entryMid
   199→
   200→      // Score all 6 layers
   201→      // Macro: build macro data from real fetched data (fearGreed, funding, etc.)
   202→      // Import macro functions for real data
   203→      const { getFearGreedIndex, getMoonPhase, getEconomicEvents } = await import('@/lib/macro')
   204→      const { getGlobalData } = await import('@/lib/coingecko')
   205→      const [fearGreedData, globalData] = await Promise.all([
   206→        getFearGreedIndex(),
   207→        getGlobalData(),
   208→      ])
   209→      const moonPhase = getMoonPhase()
   210→      const economicEvents = getEconomicEvents()
   211→      
   212→      const fgValue = fearGreedData?.value || 50
   213→      const btcDom = globalData?.btcDominance || 54
   214→      const btcDomTrend: 'UP' | 'DOWN' | 'FLAT' = btcDom > 52 ? 'UP' : 'DOWN'
   215→      const usdtTrend: 'UP' | 'DOWN' | 'FLAT' = fgValue < 40 ? 'UP' : 'DOWN'
   216→      
   217→      let fgLabel: 'Extreme Fear' | 'Fear' | 'Neutral' | 'Greed' | 'Extreme Greed' = 'Neutral'
   218→      if (fgValue < 25) fgLabel = 'Extreme Fear'
   219→      else if (fgValue < 45) fgLabel = 'Fear'
   220→      else if (fgValue < 55) fgLabel = 'Neutral'
   221→      else if (fgValue < 75) fgLabel = 'Greed'
   222→      else fgLabel = 'Extreme Greed'
   223→      
   224→      const macroData = {
   225→        btcDominance: btcDom,
   226→        btcDominanceTrend: btcDomTrend,
   227→        usdtDominance: 100 - btcDom - (globalData?.ethDominance || 17),
   228→        usdtDominanceTrend: usdtTrend,
   229→        otherBtcRatio: btcDom > 0 ? (100 - btcDom) / btcDom : 0.5,
   230→        otherBtcTrend: 'UP' as const,
   231→        fearGreed: fgValue,
   232→        fearGreedLabel: fgLabel,
   233→        fundingAggregate: funding?.fundingRate || 0.0008,
   234→        openInterestBtc: oi?.openInterestValue || 18e9,
   235→        longShortRatio: lsr?.longShortRatio || 1.0,
   236→        moonPhase,
   237→        economicEvents,
   238→        macroVerdict: 'NEUTRAL' as Bias, // Will be set from scoreMacro result below
   239→      }
   240→      const macroScore = scoreMacro(macroData)
   241→      // Use the real macro verdict from scoreMacro, not user's bias
   242→      macroData.macroVerdict = macroScore.bias
   243→      const narrativeScore = scoreNarrative(null) // No narrative context in this MVP
   244→      const fundamentalScore = scoreFundamental(fundamental)
   245→      const technicalScore = scoreTechnical(timeframeData, bias)
   246→      const checklistScore = scoreChecklist(checklist)
   247→      const riskScore = scoreRisk(avgRr, settings.riskPerTrade, openPositions, settings.maxConcurrentPositions, dailyDD, settings.dailyDrawdownLimit)
   248→
   249→      const totalScore = calculateTotalScore({
   250→        macro: macroScore,
   251→        narrative: narrativeScore,
   252→        fundamental: fundamentalScore,
   253→        technical: technicalScore,
   254→        checklist: checklistScore,
   255→        risk: riskScore,
   256→      })
   257→
   258→      // Build the trade setup object
   259→      const setup: TradeSetup = {
   260→        id: Math.random().toString(36).slice(2),
   261→        symbol,
   262→        bias,
   263→        setupType,
   264→        confidence: totalScore.totalScore,
   265→        createdAt: new Date().toISOString(),
   266→        status: checklistScore.allRequiredChecked && totalScore.totalScore >= 70 ? 'READY' : 'WAIT_TRIGGER',
   267→        timeframeAnalysis: timeframeData,
   268→        fibonacci: fib,
   269→        entry: {
   270→          zone: entryZone,
   271→          trigger: entryTrigger,
   272→          triggerPrice: entryMid,
   273→        },
   274→        stopLoss: {
   275→          price: stopLoss,
   276→          reason: bias === 'LONG' 
   277→            ? 'Below Fib 0.786 + structure invalidation' 
   278→            : 'Above Fib 0.786 + structure invalidation',
   279→        },
   280→        takeProfits: {
   281→          tp1: { price: tps.tp1, percent: 50, rr: tp1Rr },
   282→          tp2: { price: tps.tp2, percent: 30, rr: tp2Rr },
   283→          tp3: { price: tps.tp3, percent: 20, rr: tp3Rr },
   284→        },
   285→        rrAverage: avgRr,
   286→        positionSize: {
   287→          capital: settings.capital,
   288→          riskPercent: settings.riskPerTrade,
   289→          riskAmount,
   290→          quantity,
   291→          nominalValue,
   292→        },
   293→        invalidityConditions: [
   294→          `4H close ${bias === 'LONG' ? 'below' : 'above'} ${stopLoss.toFixed(currentPrice < 1 ? 5 : 2)} (structure broken)`,
   295→          funding && Math.abs(funding.fundingRate) > 0.0005 ? `Funding rate spikes > |0.05%|` : 'Funding rate stays neutral',
   296→          'Narrative dies (top 5 narrative coins all -10% in 24h)',
   297→          'BTC drops >5% in 24h (macro invalidation)',
   298→        ],
   299→        checklist,
   300→        alerts: [
   301→          { id: '1', type: 'INFO', condition: `Price enters entry zone (${entryZone.lower.toFixed(currentPrice < 1 ? 5 : 2)} - ${entryZone.upper.toFixed(currentPrice < 1 ? 5 : 2)})`, triggered: false },
   302→          { id: '2', type: 'ACTION', condition: entryTrigger, triggered: false },
   303→          { id: '3', type: 'WARNING', condition: `Price approaches SL (${stopLoss.toFixed(currentPrice < 1 ? 5 : 2)})`, triggered: false },
   304→          { id: '4', type: 'CRITICAL', condition: `SL hit`, triggered: false },
   305→          { id: '5', type: 'INFO', condition: `TP1 hit (${tps.tp1.toFixed(currentPrice < 1 ? 5 : 2)}) → move SL to BE`, triggered: false },
   306→          { id: '6', type: 'INFO', condition: `TP2 hit → trailing stop active`, triggered: false },
   307→          { id: '7', type: 'CRITICAL', condition: `Invalidity condition triggered → EXIT`, triggered: false },
   308→        ],
   309→        coinFundamental: fundamental,
   310→        narrative: null,
   311→      }
   312→
   313→      setResult({
   314→        setup,
   315→        score: totalScore,
   316→        candles4H: candlesByTf['4H'] || [],
   317→        scoreNotes: {
   318→          macro: macroScore.notes,
   319→          narrative: narrativeScore.notes,
   320→          fundamental: fundamentalScore.notes,
   321→          technical: technicalScore.notes,
   322→          checklist: checklistScore.notes,
   323→          risk: riskScore.notes,
   324→        },
   325→      })
   326→    } catch (err) {
   327→      console.error(err)
   328→      setError(err instanceof Error ? err.message : 'Analysis failed')
   329→    } finally {
   330→      setLoading(false)
   331→    }
   332→  }

   334→  const [savedSetupId, setSavedSetupId] = useState<string | null>(null)
   335→  
   336→  function saveSetup() {
   337→    if (!result) return
   338→    if (savedSetupId === result.setup.id) return // Prevent duplicate save
   339→    addSetup(result.setup)
   340→    setSavedSetupId(result.setup.id)
   341→    pushAlert({
   342→      type: 'INFO',
   343→      title: 'Setup Saved',
   344→      message: `${result.setup.symbol} ${result.setup.bias} setup saved with confidence ${result.score.totalScore.toFixed(1)}`,
   345→    })
   346→  }

   351→    ))
   352→  }

   376→              />

   378→            <div>

   380→              <Select value={bias} onValueChange={(v) => setBias(v as Bias)}>

   399→                onChange={(e) => setOpenPositions(parseInt(e.target.value) || 0)}

   401→                className="bg-zinc-950 border-zinc-700 text-sm"
   402→              />
   403→            </div>
   404→            <div>
   405→              <Label className="text-xs text-zinc-400">Daily DD (%)</Label>

   408→                value={dailyDD}
   409→                onChange={(e) => setDailyDD(parseFloat(e.target.value) || 0)}

   414→          </div>
   415→

   417→            onClick={analyze}

   419→            className="w-full"
   420→            size="sm"

   422→            {loading ? (

   441→            {error}

   443→        </Card>
   444→      )}
   445→
   446→      {/* Results */}
   447→      {result && (
   448→        <Tabs defaultValue="summary" className="w-full">
   449→          <TabsList className="grid w-full grid-cols-5 h-8">

   451→            <TabsTrigger value="technical" className="text-xs">Technical</TabsTrigger>
   452→            <TabsTrigger value="chart" className="text-xs">Chart</TabsTrigger>
   453→            <TabsTrigger value="checklist" className="text-xs">Checklist</TabsTrigger>
   454→            <TabsTrigger value="alerts" className="text-xs">Alerts</TabsTrigger>
   455→          </TabsList>
   456→
   457→          {/* SUMMARY TAB */}
   458→          <TabsContent value="summary" className="space-y-3 mt-2">
   459→            {/* Confidence Badge */}
   460→            <Card className={`border-l-4 ${
   461→              result.score.totalScore >= 80 ? 'border-l-emerald-500 bg-emerald-950/30' :
   462→              result.score.totalScore >= 70 ? 'border-l-yellow-500 bg-yellow-950/20' :

   471→                    <div className="text-xs text-zinc-500">{result.score.confidence}</div>
   472→                  </div>
   473→                  <Badge variant={result.setup.status === 'READY' ? 'default' : 'secondary'} className={
   474→                    result.setup.status === 'READY' ? 'bg-emerald-600' : ''

   476→                    {result.setup.status === 'READY' ? '✅ READY' : '⏳ WAIT'}

   483→                  <LayerScoreCard label="L2 Narrative" score={result.score.layer2Narrative} notes={result.scoreNotes.narrative} />
   484→                  <LayerScoreCard label="L3 Fund" score={result.score.layer3Fundamental} notes={result.scoreNotes.fundamental} />
