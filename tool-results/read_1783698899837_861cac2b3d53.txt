    43→export function SetupFinder() {
    44→  const [symbol, setSymbol] = useState('BTCUSDT')
    45→  const [bias, setBias] = useState<Bias>('LONG')
    46→  const [loading, setLoading] = useState(false)
    47→  const [result, setResult] = useState<SetupResult | null>(null)
    48→  const [error, setError] = useState<string | null>(null)
    49→  const [allCoins, setAllCoins] = useState<CoinMarketData[]>([])
    50→  const [checklist, setChecklist] = useState(defaultChecklist())
    51→  const [openPositions, setOpenPositions] = useState(0)
    52→  const [dailyDD, setDailyDD] = useState(0)
    53→
    54→  const { settings, addSetup, pushAlert, setups } = useAppStore()
    55→
    56→  // Load top coins for autocomplete
    57→  useEffect(() => {
    58→    async function loadCoins() {
    59→      const coins = await getTopCoins(100)
    60→      setAllCoins(coins)
    61→    }
    62→    loadCoins()
    63→  }, [])
    64→
    65→  async function analyze() {
    66→    setLoading(true)
    67→    setError(null)
    68→    setResult(null)
    69→    try {
    70→      // Fetch data for all timeframes
    71→      const timeframeData: Record<Timeframe, TechnicalAnalysis> = {} as Record<Timeframe, TechnicalAnalysis>
    72→      const candlesByTf: Record<Timeframe, Candle[]> = {} as Record<Timeframe, Candle[]>
    73→      
    74→      await Promise.all(
    75→        TIMEFRAMES.map(async (tf) => {
    76→          const candles = await getKlines(symbol, tf, 300)
    77→          candlesByTf[tf] = candles
    78→          // Always analyze, even with few candles (analyzeTimeframe handles edge cases)
    79→          timeframeData[tf] = analyzeTimeframe(candles, tf)
    80→        })
    81→      )
    82→
    83→      // Check we have at least 1H and 4H data (the entry timeframes)
    84→      if (!timeframeData['1H'] || !timeframeData['4H']) {
    85→        throw new Error('Failed to fetch 1H/4H data. Check symbol.')
    86→      }
    87→
    88→      // Fetch additional data
    89→      const [ticker, funding, oi, lsr] = await Promise.all([
    90→        getTicker24h(symbol),
    91→        getFundingRate(symbol),
    92→        getOpenInterest(symbol.replace('USDT', '') + 'USDT'),
    93→        getLongShortRatio(symbol.replace('USDT', '') + 'USDT'),
    94→      ])
    95→
    96→      // Find coin fundamental (from CoinGecko)
    97→      const coinData = allCoins.find(c => 
    98→        c.symbol + 'USDT' === symbol || c.symbol === symbol.replace('USDT', '')
    99→      )
   100→      
   101→      const fundamental: CoinFundamental = {
   102→        symbol: symbol.replace('USDT', ''),
   103→        name: coinData?.name || symbol,
   104→        marketCap: coinData?.marketCap || 0,
   105→        marketCapRank: coinData?.marketCapRank || 999,
   106→        volume24h: ticker?.quoteVolume || coinData?.totalVolume || 0,
   107→        volToMcap: coinData ? (ticker?.quoteVolume || coinData.totalVolume) / coinData.marketCap : 0,
   108→        exchangeListings: 8,
   109→        tokenomics: {
   110→          circulating: coinData?.circulatingSupply || 0,
   111→          total: coinData?.totalSupply || 0,
   112→          fdv: coinData && coinData.totalSupply ? coinData.currentPrice * coinData.totalSupply : 0,
   113→          fdvToMcap: coinData && coinData.totalSupply && coinData.marketCap > 0 
   114→            ? (coinData.currentPrice * coinData.totalSupply) / coinData.marketCap 
   115→            : 1,
   116→          nextUnlock: null,
   117→        },
   118→        activeAddresses30d: 0,
   119→        activeAddressesTrend: 'FLAT',
   120→        devActivity: 50,
   121→        newsSentiment: 50,
   122→        verdict: 'PASS',
   123→      }
   124→
   125→      // Determine setup type based on technical analysis
   126→      const daily = timeframeData['1D']
   127→      const h4 = timeframeData['4H']
   128→      
   129→      let setupType: SetupType = 'TREND_PULLBACK'
   130→      if (daily.adx < 20) setupType = 'MEAN_REVERSION'
   131→      else if (h4.bollinger.squeeze && daily.bollinger.squeeze) setupType = 'BREAKOUT'
   132→
   133→      // Calculate Fibonacci levels
   134→      const recentCandles = candlesByTf['4H'].slice(-50)
   135→      const swingLow = Math.min(...recentCandles.map(c => c.low))
   136→      const swingHigh = Math.max(...recentCandles.map(c => c.high))
   137→      const fib = fibonacci(swingLow, swingHigh)
   138→
   139→      // Calculate entry zone, SL, TP based on bias
   140→      const currentPrice = timeframeData['1H'].price
   141→      let entryZone: { lower: number; upper: number }
   142→      let stopLoss: number
   143→      let tps: { tp1: number; tp2: number; tp3: number }
   144→      let entryTrigger: string
   145→
   146→      if (bias === 'LONG') {
   147→        // Long setup: entry at pullback to fib 0.5-0.618 + EMA50 confluence
   148→        const fib618 = fib.levels['0.618']
   149→        const fib500 = fib.levels['0.500']
   150→        const ema50_4h = h4.ema50
   151→        entryZone = {
   152→          lower: Math.min(fib618, ema50_4h * 0.99),
   153→          upper: Math.max(fib500, ema50_4h * 1.01),
   154→        }
   155→        // SL below fib 0.786 or recent swing low
   156→        stopLoss = Math.min(fib.levels['0.786'], swingLow * 0.98)
   157→        // TPs at previous high + fib extensions
   158→        tps = {
   159→          tp1: swingHigh,
   160→          tp2: fib.extensions['1.272'],
   161→          tp3: fib.extensions['1.618'],
   162→        }
   163→        entryTrigger = `1H close above ${((entryZone.lower + entryZone.upper) / 2 * 1.005).toFixed(currentPrice < 1 ? 5 : 2)} with volume 1.5x avg`
   164→      } else {
   165→        // Short setup: entry at pullback to fib 0.5-0.618 (from highs)
   166→        const fib618 = fib.levels['0.618']
   167→        const fib500 = fib.levels['0.500']
   168→        const ema50_4h = h4.ema50
   169→        entryZone = {
   170→          lower: Math.min(fib500, ema50_4h * 0.99),
   171→          upper: Math.max(fib618, ema50_4h * 1.01),
   172→        }
   173→        stopLoss = Math.max(fib.levels['0.786'], swingHigh * 1.02)
   174→        tps = {
   175→          tp1: swingLow,
   176→          tp2: swingLow - (swingHigh - swingLow) * 0.272,
   177→          tp3: swingLow - (swingHigh - swingLow) * 0.618,
   178→        }
   179→        entryTrigger = `1H close below ${((entryZone.lower + entryZone.upper) / 2 * 0.995).toFixed(currentPrice < 1 ? 5 : 2)} with volume 1.5x avg`
   180→      }
   181→
   182→      const entryMid = (entryZone.lower + entryZone.upper) / 2
   183→      const risk = Math.abs(entryMid - stopLoss)
   184→      const tp1Rr = Math.abs(tps.tp1 - entryMid) / risk
   185→      const tp2Rr = Math.abs(tps.tp2 - entryMid) / risk
   186→      const tp3Rr = Math.abs(tps.tp3 - entryMid) / risk
   187→      const avgRr = (tp1Rr * 0.5 + tp2Rr * 0.3 + tp3Rr * 0.2)
   188→
   189→      // Position size
   190→      const riskAmount = settings.capital * (settings.riskPerTrade / 100)
   191→      const usdToIdr = 15800 // approximate
   192→      const usdRisk = riskAmount / usdToIdr
   193→      const quantity = usdRisk / risk
   194→      const nominalValue = quantity * entryMid
   195→
   196→      // Score all 6 layers
   197→      // Macro: build simplified macro data
   198→      const macroData = {
   199→        btcDominance: 54,
   200→        btcDominanceTrend: 'DOWN' as const,
   201→        usdtDominance: 5.8,
   202→        usdtDominanceTrend: 'DOWN' as const,
   203→        otherBtcRatio: 0.42,
   204→        otherBtcTrend: 'UP' as const,
   205→        fearGreed: 32,
   206→        fearGreedLabel: 'Fear' as const,
   207→        fundingAggregate: funding?.fundingRate || 0.0008,
   208→        openInterestBtc: oi?.openInterestValue || 18e9,
   209→        longShortRatio: lsr?.longShortRatio || 1.0,
   210→        moonPhase: { phase: 'Waxing Crescent', emoji: '🌒', bullBearBias: 'BULLISH' as const, daysToNext: 5 },
   211→        economicEvents: [],
   212→        macroVerdict: bias as Bias,
   213→      }
   214→      const macroScore = scoreMacro(macroData)
   215→      const narrativeScore = scoreNarrative(null) // No narrative context in this MVP
   216→      const fundamentalScore = scoreFundamental(fundamental)
   217→      const technicalScore = scoreTechnical(timeframeData, bias)
   218→      const checklistScore = scoreChecklist(checklist)
   219→      const riskScore = scoreRisk(avgRr, settings.riskPerTrade, openPositions, settings.maxConcurrentPositions, dailyDD, settings.dailyDrawdownLimit)
   220→
   221→      const totalScore = calculateTotalScore({
   222→        macro: macroScore,
   223→        narrative: narrativeScore,
   224→        fundamental: fundamentalScore,
   225→        technical: technicalScore,
   226→        checklist: checklistScore,
   227→        risk: riskScore,
   228→      })
   229→
   230→      // Build the trade setup object
   231→      const setup: TradeSetup = {
   232→        id: Math.random().toString(36).slice(2),
   233→        symbol,
   234→        bias,
   235→        setupType,
   236→        confidence: totalScore.totalScore,
   237→        createdAt: new Date().toISOString(),
   238→        status: checklistScore.allRequiredChecked && totalScore.totalScore >= 70 ? 'READY' : 'WAIT_TRIGGER',
   239→        timeframeAnalysis: timeframeData,
   240→        fibonacci: fib,
   241→        entry: {
   242→          zone: entryZone,
   243→          trigger: entryTrigger,
   244→          triggerPrice: entryMid,
   245→        },
   246→        stopLoss: {
   247→          price: stopLoss,
   248→          reason: bias === 'LONG' 
   249→            ? 'Below Fib 0.786 + structure invalidation' 
   250→            : 'Above Fib 0.786 + structure invalidation',
   251→        },
   252→        takeProfits: {
   253→          tp1: { price: tps.tp1, percent: 50, rr: tp1Rr },
   254→          tp2: { price: tps.tp2, percent: 30, rr: tp2Rr },
   255→          tp3: { price: tps.tp3, percent: 20, rr: tp3Rr },
   256→        },
   257→        rrAverage: avgRr,
   258→        positionSize: {
   259→          capital: settings.capital,
   260→          riskPercent: settings.riskPerTrade,
   261→          riskAmount,
   262→          quantity,
   263→          nominalValue,
   264→        },
   265→        invalidityConditions: [
   266→          `4H close ${bias === 'LONG' ? 'below' : 'above'} ${stopLoss.toFixed(currentPrice < 1 ? 5 : 2)} (structure broken)`,
   267→          funding && Math.abs(funding.fundingRate) > 0.0005 ? `Funding rate spikes > |0.05%|` : 'Funding rate stays neutral',
   268→          'Narrative dies (top 5 narrative coins all -10% in 24h)',
   269→          'BTC drops >5% in 24h (macro invalidation)',
   270→        ],
   271→        checklist,
   272→        alerts: [
   273→          { id: '1', type: 'INFO', condition: `Price enters entry zone (${entryZone.lower.toFixed(currentPrice < 1 ? 5 : 2)} - ${entryZone.upper.toFixed(currentPrice < 1 ? 5 : 2)})`, triggered: false },
   274→          { id: '2', type: 'ACTION', condition: entryTrigger, triggered: false },
   275→          { id: '3', type: 'WARNING', condition: `Price approaches SL (${stopLoss.toFixed(currentPrice < 1 ? 5 : 2)})`, triggered: false },
   276→          { id: '4', type: 'CRITICAL', condition: `SL hit`, triggered: false },
   277→          { id: '5', type: 'INFO', condition: `TP1 hit (${tps.tp1.toFixed(currentPrice < 1 ? 5 : 2)}) → move SL to BE`, triggered: false },
   278→          { id: '6', type: 'INFO', condition: `TP2 hit → trailing stop active`, triggered: false },
   279→          { id: '7', type: 'CRITICAL', condition: `Invalidity condition triggered → EXIT`, triggered: false },
   280→        ],
   281→        coinFundamental: fundamental,
   282→        narrative: null,
   283→      }
   284→
   285→      setResult({
   286→        setup,
   287→        score: totalScore,
   288→        candles4H: candlesByTf['4H'] || [],
   289→        scoreNotes: {
   290→          macro: macroScore.notes,
   291→          narrative: narrativeScore.notes,
   292→          fundamental: fundamentalScore.notes,
   293→          technical: technicalScore.notes,
   294→          checklist: checklistScore.notes,
   295→          risk: riskScore.notes,
   296→        },
   297→      })
   298→    } catch (err) {
   299→      console.error(err)
   300→      setError(err instanceof Error ? err.message : 'Analysis failed')
   301→    } finally {
   302→      setLoading(false)
   303→    }
   304→  }

   306→  function saveSetup() {

   308→    addSetup(result.setup)

   310→      type: 'INFO',
   311→      title: 'Setup Saved',
   312→      message: `${result.setup.symbol} ${result.setup.bias} setup saved with confidence ${result.score.totalScore.toFixed(1)}`,
   313→    })
   314→  }

   331→            Coin Analysis Input
   332→          </CardTitle>
   333→        </CardHeader>
   334→        <CardContent className="space-y-3">
   335→          <div className="grid grid-cols-2 gap-3">

   337→              <Label htmlFor="symbol" className="text-xs text-zinc-400">Symbol (Binance)</Label>
   338→              <Input
   339→                id="symbol"
   340→                value={symbol}
   341→                onChange={(e) => setSymbol(e.target.value.toUpperCase())}
   342→                placeholder="BTCUSDT"
   343→                className="bg-zinc-950 border-zinc-700 text-sm"
   344→              />
   345→            </div>
   346→            <div>
   347→              <Label className="text-xs text-zinc-400">Trade Bias</Label>
   348→              <Select value={bias} onValueChange={(v) => setBias(v as Bias)}>

   380→              />
   381→            </div>
   382→          </div>
   383→
   384→          <Button

   387→            className="w-full"

   389→          >

   415→      {result && (

   418→            <TabsTrigger value="summary" className="text-xs">Summary</TabsTrigger>
   419→            <TabsTrigger value="technical" className="text-xs">Technical</TabsTrigger>
   420→            <TabsTrigger value="chart" className="text-xs">Chart</TabsTrigger>
   421→            <TabsTrigger value="checklist" className="text-xs">Checklist</TabsTrigger>
   422→            <TabsTrigger value="alerts" className="text-xs">Alerts</TabsTrigger>
   423→          </TabsList>
   424→
   425→          {/* SUMMARY TAB */}
   426→          <TabsContent value="summary" className="space-y-3 mt-2">
   427→            {/* Confidence Badge */}
   428→            <Card className={`border-l-4 ${
   429→              result.score.totalScore >= 80 ? 'border-l-emerald-500 bg-emerald-950/30' :
   430→              result.score.totalScore >= 70 ? 'border-l-yellow-500 bg-yellow-950/20' :
   431→              result.score.totalScore >= 60 ? 'border-l-orange-500 bg-orange-950/20' :
   432→              'border-l-red-500 bg-red-950/20'
   433→            }`}>
   434→              <CardContent className="p-3">
   435→                <div className="flex items-center justify-between mb-2">
